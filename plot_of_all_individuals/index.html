Plot of all individuals

using Pumas, PumasPlots, StatsPlots
include("model.jl")

using Pumas PumasPlots

plot_grouped(
    groupby(DataFrame(inspect(res)), :id);
    # needs to be a large size for the plots to
    # be visible.  Try to maintain a square aspect
    # ratio.
    size = (2000, 2000),
    # we don't need a legend
    legend = :none,
    # link all axes together
    link = :all,
    # the figures should be laid out tightly
    margin = 0Plots.px,
    # these fontsizes are based on personal preference,
    # but should be fairly large, so that they can be
    # seen in PDFs.
    tickfontsize = 14,
    titlefontsize = 15
) do subdf
    @df subdf plot(:time, :dv_ipred)
end
